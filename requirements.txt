# Google Agent Development Kit (ADK) -- required by the `adk/` layer
# (`python chat.py` / `adk web`). Without it `import adk` raises ModuleNotFoundError.
# The [db] extra pulls SQLAlchemy so DatabaseSessionService (SQLite) works for
# persistent sessions across restarts. Without it the REPL falls back to in-memory.
#
# >=2.3.0 is a hard requirement, not just "latest and greatest": adk/agent.py
# passes on_tool_error_callback/on_model_error_callback to Agent(...), which
# older ADK releases (confirmed: 1.14.1) reject at startup with a pydantic
# "Extra inputs are not permitted" ValidationError -- the agent never even
# imports. A permissive ">=1.0.0" let a fresh `pip install` silently resolve
# to an incompatible older release (some other dependency's transitive
# constraint backed pip's resolver down from PyPI's actual latest); this
# only surfaced as a Docker container crashing on startup, since a
# long-lived local dev venv had already been on 2.3.0 from before.
#
# Deliberately WITHOUT the [extensions] extra: its only relevant piece is
# litellm's ADK integration (google.adk.models.lite_llm.LiteLlm, used for
# non-Gemini providers -- see utils/model_config.py), but that module's own
# imports are just `litellm` (already a separate dependency below) plus
# stdlib/pydantic -- confirmed by reading lite_llm.py's import list directly.
# [extensions] itself bundles crewai, langgraph, llama-index, kubernetes,
# docker, google-cloud-firestore and more -- none used anywhere in this
# project -- and added ~1GB+ to the Docker image for zero functional benefit.
google-adk[db]>=2.3.0

# Async SQLite driver for DatabaseSessionService persistence
# (used when POWERBI_SESSION_DB_URL=sqlite+aiosqlite:///...). Optional but
# recommended; without it the REPL falls back to InMemorySessionService.
aiosqlite>=0.20.0

# ASGI server + health endpoint (adk/server.py). uvicorn serves the FastAPI
# app with GET /health for liveness probes; fastapi pulls in starlette/httpx.
uvicorn>=0.30.0
fastapi>=0.110.0

# (Optional) LLM client for richer DAX/report generation.
# The system works fully offline with deterministic rules, but if an
# API key is provided via env var, agents will enhance their output.
# google-generativeai and openai are intentionally NOT pinned hard so
# the project still installs without them.

# Provider-agnostic LLM completion for the pipeline layer (utils/model_config.py).
# Lazily imported (same pattern as google-generativeai above) -- without it,
# the LLM-enhancement paths just no-op back to the deterministic fallback.
# Enables Anthropic/OpenAI (and any future provider) via one shared interface
# instead of each agent hardcoding a Google genai.Client() call.
litellm>=1.0.0
